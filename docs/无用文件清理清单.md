## 七、执行清理的PowerShell命令

```powershell
# 设置路径
$root = "d:\桌面\大二下学期期末考试\学年设计\swu-cq-house-price-analysis"

# 1. 删除tmp目录
Remove-Item -Path "$root\tmp" -Recurse -Force

# 2. 删除根目录临时文件
Remove-Item -Path "$root\check_house_age.py" -Force

# 3. 删除scripts一次性脚本
$scriptsToDelete = @(
    "crawl_probe.py",
    "verify_incremental_snapshot.py",
    "verify_scheduled_incremental_run.py",
    "normalize_fang_liangjiang_district.py",
    "purge_legacy_data.py",
    "repair_anjuke_polluted_addresses.py",
    "run_full_crawl.py",
    "local_demo_smoke.py",
    "full_functional_test.js",
    "ui_acceptance_check.js",
    "run_ui_acceptance.ps1",
    "start_local_dev.ps1"
)

foreach ($file in $scriptsToDelete) {
    $path = "$root\scripts\$file"
    if (Test-Path $path) {
        Remove-Item -Path $path -Force
        Write-Host "已删除: $file"
    }
}

Write-Host "清理完成！"
```
